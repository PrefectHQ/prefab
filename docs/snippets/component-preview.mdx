export const ComponentPreview = ({ json, height, children }) => {
  const hostRef = React.useRef(null);
  const handleRef = React.useRef(null);

  React.useEffect(() => {
    let cancelled = false;
    const host = hostRef.current;
    if (!host) return;

    var loadModule = new Function("u", "return import(u)");
    loadModule("/embed.js").then(({ mountPreview }) => {
      if (cancelled) return;
      var dark = document.documentElement.classList.contains("dark");
      handleRef.current = mountPreview(host, json, { dark: dark });
    });

    var observer = new MutationObserver(() => {
      var dark = document.documentElement.classList.contains("dark");
      if (handleRef.current) handleRef.current.setDark(dark);
    });
    observer.observe(document.documentElement, {
      attributes: true,
      attributeFilter: ["class"],
    });

    return () => {
      cancelled = true;
      observer.disconnect();
      if (handleRef.current) {
        handleRef.current.unmount();
        handleRef.current = null;
      }
    };
  }, []);

  return (
    <Card>
      <div ref={hostRef} />
      <div style={{ marginBottom: "-2rem" }}>{children}</div>
    </Card>
  );
};
